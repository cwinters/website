Use the mixin method

  • Mixin just like the real thing
#!/usr/bin/perl

use strict;
use SearchCommonTwo qw( initialize output );

initialize();
my ( $title, $num ) = @ARGV;
die "Usage $0 title issue-num\n" unless ( $title and $num );
my $comic = CBG::Comic->fetch_by_issue( $title, $num );
output( ( $comic ) ? [ $comic ] : [] );
