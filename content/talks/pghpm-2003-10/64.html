Simple ruleset example

  • Those dates are butt-ugly
  • We could just create DateTime objects as we go...
  • ...but that's what frameworks are for
$spops = {
  comic => {
     class           => 'CBG::Comic',
     code_class      => 'CBG::ComicCode',
     isa             => [ 'CBG::Object', 'SPOPS::DBI::Pg', 'SPOPS::DBI' ],
     rules_from      => [ 'SPOPS::Tool::DBI::DiscoverField' ],
     field           => [],
     field_discover  => 'yes',
     id_field        => 'comic_id',
     no_insert       => 'comic_id',
     no_update       => 'comic_id',
     increment_field => 1,
     sequence_name   => 'comic_seq',
     base_table      => 'comic',
     links_to        => {
          'CBG::ComicPrice' => 'price_history',
     },
  },
  price => {
     class           => 'CBG::ComicPrice',
     isa             => [ 'CBG::Object', 'SPOPS::DBI::Pg', 'SPOPS::DBI' ],
     rules_from      => [ 'SPOPS::Tool::DBI::DiscoverField',
                          'SPOPS::Tool::DateConvert' ],
     field           => [],
     field_discover  => 'yes',
     id_field        => 'price_history_id',
     no_insert       => 'price_history_id',
     no_update       => 'price_history_id',
     increment_field => 1,
     sequence_name   => 'price_history_seq',
     base_table      => 'price_history',
     convert_date_field  => [ 'change_date' ],
     convert_date_class  => 'DateTime',
     convert_date_format => '%Y-%m-%d %H:%M:%S',
     has_a           => {
          'CBG::Comic' => 'comic_id',
     },
  },
};
