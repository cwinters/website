  • Basic schema plus sequence for Pg:
CREATE TABLE comic (
  comic_id     int not null primary key,
  publisher    varchar(30) not null,
  title        varchar(50) not null,
  issue_num    varchar(8) not null,
  condition    varchar(5) not null,
  price        numeric(10,2),
  writer       varchar(50) null,
  penciller    varchar(50) null,
  inker        varchar(50) null,
  notes        varchar(255) null,
  acquire_date timestamp,
  UNIQUE( title, issue_num )
);

CREATE SEQUENCE comic_seq;
