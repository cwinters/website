OT: How SQLite implements 'TYPE'

  • Not something to fix, just be aware for DBD::SQLite
  • What does the following script print out?
#!/usr/bin/perl

use strict;
use DBI;

my $dbh = DBI->connect( 'DBI:SQLite:dbname=spops.db', '', '' )
                    || die $DBI::errstr;
$dbh->{RaiseError} = 1;
my $sth = $dbh->prepare( 'SELECT * FROM job' );
$sth->execute;
print "Types: ", join( ', ', @{ $sth->{TYPE} } ), "\n";
print "Names: ", join( ', ', @{ $sth->{NAME} } ), "\n";
