  • Now process the configuration
  • Process == generate code for classes and evaluate
  • Only one module required to 'use'
#!/usr/bin/perl

use strict;
use SPOPS::Initialize;
use vars qw( $job );

do 'job_map.pl';
my $classes = SPOPS::Initialize->process({ config => $job });
print "I created the following classes: ",
      join( ', ', @{ $classes } ), "\n";
