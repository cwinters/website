One-minute intro (5)

  • And fetch the objects you've created
#!/usr/bin/perl

use strict;
use SPOPS::Initialize;
use vars qw( $job );

do 'job_map.pl';
SPOPS::Initialize->process({ config => $job });

my $all_items = My::Job->fetch_group;
print "All items:\n", join( "\n", map { show_job( $_ ) } @{ $all_items } ), "\n";

my $single = My::Job->fetch(1);
print "Single item: \n", show_job( $single ), "\n";

my $not = My::Job->fetch(15);
print "Result of nonexistent item defined? ", ( defined $not ) ? 'yes' : 'no', "\n";

sub show_job {
    my ( $job ) = @_;
    return "    $job->{id}: $job->{name} is a $job->{occupation}"
}
