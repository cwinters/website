Declaring security

  • Configuration specifies what happens on object creation
  • Key 'creation_security' used
$job = {
  job => {
     class      => 'My::Job',
     isa        => [ 'SPOPS::DBI::SQLite', 'SPOPS::DBI' ],
     rules_from => [ 'SPOPS::Tool::DBI::Datasource' ],
     field      => [ 'id', 'name', 'occupation' ],
     id_field   => 'id',
     base_table => 'job',
     dbi_config => { dsn      => 'DBI:SQLite:dbname=spops.db',
                     username => '',
                     password => '' },
     creation_security => {
        world => 'READ',
        group => { 3 => 'WRITE' },
     },
  },
};
