Want to view individual comic

  • But who wants to remember IDs?
  • 'UNIQUE' constraint on title + issue number
  • Create a class and method to fetch
  • Add class to 'code_class' (mixin)
package CBG::ComicCode;

use strict;

sub fetch_by_issue {
    my ( $class, $title, $issue_number ) = @_;
    my $found = $class->fetch_group(
                    { where => 'title = ? AND issue_num = ?',
                      value => [ $title, $issue_number ] });
    return ( ref $found eq 'ARRAY' and scalar @{ $found } )
             ? $found->[0] : undef;
}

1;
