Create some query scripts

  • CBG is lazy and doesn't want a GUI
  • Simple one-offs for title, artist, publisher searches
  • Example: publisher
#!/usr/bin/perl

use strict;
use SearchCommon qw( initialize output );

initialize();
my ( $publisher ) = @ARGV;
die "Usage $0 publisher\n" unless ( $publisher );
my $comics = CBG::Comic->fetch_group(
                    { where => 'publisher LIKE ?',
                      value => [ "%$publisher%" ],
                      order => 'publisher, title, issue_num' });
output( $comics );
