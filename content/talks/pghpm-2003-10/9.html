One-minute intro (4)

  • Now just use the class and create objects
#!/usr/bin/perl

use strict;
use SPOPS::Initialize;
use vars qw( $job );

-f 'spops.db' && unlink( 'spops.db' );
do 'job_map.pl';
SPOPS::Initialize->process({ config => $job });
My::Job->global_datasource_handle()->do( join( '', <DATA> ) );

print "Class is using table: ", My::Job->table_name, "\n";

my $one = My::Job->new({
    id => 1, name => 'Chuck', occupation => 'Football kicker'
})->save();
print "Is first object saved? ", ( $one->is_saved ) ? 'yes' : 'no', "\n";

my $two = My::Job->new();
$two->{id}         = 2;
$two->{name}       = 'Schroeder';
$two->{occupation} = 'Piano man';
$two->save();
print "Is second object saved? ", ( $one->is_saved ) ? 'yes' : 'no';

__DATA__
CREATE TABLE job (
   id         int not null primary key,
   name       varchar(40) not null,
   occupation varchar(40) not null
)
