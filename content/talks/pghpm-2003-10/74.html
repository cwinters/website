Tracking global resources

  • Database handle example of general pattern
  • Constrained resources defined in parent w/get and set access
  • Examples: current user, current group
package ParentUserContext;

use strict;

my ( $USER, $GROUPS );

sub get_user {
    return $USER;
}

sub set_user {
    my ( $self, $user ) = @_;
    return unless ( $user );
    $USER = $user;
}

sub get_groups {
    return $GROUPS;
}

sub set_groups {
    my ( $self, $groups ) = @_;
    return unless ( $groups );
    $GROUPS = $groups;
}

1;
