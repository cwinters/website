Get related objects

  • Now we can grab all prices for a comic
#!/usr/bin/perl

use strict;
use SearchCommonThree qw( initialize output );

initialize();
my ( $title, $num ) = @ARGV;
die "Usage $0 title issue-num\n" unless ( $title and $num );
my $comic = CBG::Comic->fetch_by_issue( $title, $num );
output( ( $comic ) ? [ $comic ] : [] );

my $prices = $comic->price;
foreach my $price ( sort by_date @{ $prices } ) {
    print "  $price->{change_date}: \$$price->{price}\n";
}

sub by_date {
    return $a->{change_date} cmp $b->{change_date};
}
