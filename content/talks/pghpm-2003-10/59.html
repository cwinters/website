Declare parent in 'isa'

  • Added a new class to 'isa': 'CBG::Object'
  • We subclass this just like we've used @ISA
  • Manages our database connection for us
package CBG::Object;

use strict;
use vars qw( $dbi_config );
use DBI;

do 'comic_dbi_config.pl';

my ( $DBH );

sub global_datasource_handle {
    return $DBH if ( $DBH );
    my ( $item ) = @_;
    $DBH = eval {
        DBI->connect( $dbi_config->{dsn},
                      $dbi_config->{user},
                      $dbi_config->{password} ) || die $DBI::errstr
    };
    if ( $@ ) {
        die "Failed to connect to database handle using inlined ",
            "configuration: $@";
    }
    return $DBH;
}

1;
