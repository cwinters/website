Side-effect Example: Create our mock class

  • First we need a mock session manager
package Mock::SessionManager;

sub new {
    my ( $class, %data ) = @_;
    return bless( { _data => \%data }, $class );
}

sub create {
    my ( $self, $session_id ) = @_;
    return ( $session_id ) ? %{ $self->{_data} } : ();
}

sub save { # no-op }

1;
