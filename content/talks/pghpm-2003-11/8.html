  • Here's the script to feed the animals
#!/usr/bin/perl

use Cat;
use CatLady;
use Dog;

{
    my @animals = (
        Cat->new( name => 'Felix' ),
        Cat->new( name => 'Krazy' ),
        Cat->new( name => 'Stray' ),
        Dog->new( name => 'Rin Tin Tin' ),
    );
    my $cat_lady = CatLady->new( name => 'Esther' );
    $cat_lady->add_observer( $_ ) for ( @animals );

    $cat_lady->feed_animals;
}
