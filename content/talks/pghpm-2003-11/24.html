Adding progress behavior (4)

  • Sample code to respond to events with a voice:
package ProgressVoice;
# ...
sub update {
    my ( $self, $processor, $action, @params ) = @_;
    if ( $action eq 'start' ) {
        my ( $start_item, $end_item ) = @params;
        $self->num_items( $end_item - $start_item );
        $self->speak( "Starting to process $num_items" );
    }
    elsif ( $action eq 'item_complete' ) {
        my ( $num_completed ) = @params;
        $self->speak(
            "Item $num_completed of ", $self->num_items, " completed " );
    }
    elsif ( $action eq 'finish' ) {
        $self->speak( "Process completed. Have a nice day." );
    }
}

1;
