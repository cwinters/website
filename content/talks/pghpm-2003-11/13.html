What's going on here? (Animal)

  • For completeness, superclass for CatLady/Cat/Dog:
package Animal;

sub new {
    my ( $class, %params ) = @_;
    return bless( \%params, $class )->init();
}

sub init { return shift }

sub talk {
    my ( $self, $words ) = @_;
    print "$self->{name} ($self->{type}): $words\n";
}

sub feed {
    my ( $self, $food ) = @_;
    $self->talk( "Thanks for the $food!" );
}

1;
