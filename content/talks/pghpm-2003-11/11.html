What's going on here? (Cat)

  • Here's the cat's class
  • Listen for the bell, beg for food
package Cat;

use base qw( Animal );

sub init {
    my ( $self ) = @_;
    $self->{type}   = 'cat';
    $self->{name} ||= 'Some Tom';
    return $self;
}

sub update {
    my ( $self, $feeder, $action ) = @_;
    if ( $action eq 'ring bell' ) {
        $feeder->beg_for_food( $self );
    }
    else {
        $self->talk( 'yawn' );
    }
}

1;
