Adding progress behavior (3)

  • Sample code to respond to events with a bar:
package ProgressBar;
# ...
sub update {
    my ( $self, $processor, $action, @params ) = @_;
    if ( $action eq 'start' ) {
        my ( $start_item, $end_item ) = @params;
        $self->_init_bar( begin => $start_item,
                          end   => $end_item );
    }
    elsif ( $action eq 'item_complete' ) {
        my ( $num_completed ) = @params;
        $self->_update_bar( completed => $num_completed );
    }
    elsif ( $action eq 'finish' ) {
        $self->_finish_bar();
    }
}

1;
