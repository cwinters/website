Adding progress behavior (1)

  • Here's the before:
package MyProcess;

use base qw( SomeClass );
# ...

sub run {
    my ( $self ) = @_;
    my $num_thingies = 0;
    eval {
        my @thingies = $self->_fetch_some_data;
        foreach my $thingy ( @thingies ) {
            $self->_do_lot_of_stuff( $thingy );
            $num_thingies++;
        }
    };
    if ( $@ ) {
        $self->_log_error( $@ );
        die "Failed to perform process: $@\n";
    }
    return $num_thingies;
}

1;
