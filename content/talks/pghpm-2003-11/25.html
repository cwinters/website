Adding progress behavior (5)

  • Put them all together and you've got fun!
#!/usr/bin/perl

use MyProcess;
use ProgressBar;
use ProgressVoice;

{
    my $process = MyProcess->new(
                      dir => '/dir/with/many/files' );
    my $bar   = ProgressBar->new();
    my $voice = ProgressVoice->new();
    $process->add_observer( $bar );
    $process->add_observer( $voice );
# Can use as of 1.02
#   $process->add_observer( $bar, $voice );
    $process->run();
}
