What's going on here? (Cat Lady)

  • Here's the cat lady's class
  • Push the bell out as observation
  • Animals can beg for food from her
package CatLady;

use base qw( Class::Observable Animal );

sub init {
    my ( $self ) = @_;
    $self->{type}   = 'Cat Lady';
    $self->{name} ||= 'Some Kook';
    return $self;
}

sub feed_animals {
    my ( $self ) = @_;
    $self->notify_observers( 'ring bell' );
}

sub beg_for_food {
    my ( $self, $animal ) = @_;
    if ( $animal->{type} eq 'cat' ) {
        $animal->feed( 'tuna' );
    }
    else {
        $self->talk( "No food for you $animal->{name}!" );
    }
}

1;
