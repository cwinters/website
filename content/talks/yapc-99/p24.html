Apache::AuthenDBI

Example of a configuration in httpd.conf


 AuthName "YAPC Authentication Demo"
 AuthType Basic
 PerlAuthenHandler		Apache::AuthenDBI
 PerlSetVar Auth_DBI_data_source 'dbi:Sybase:server=yapc;database=web'
 PerlSetVar Auth_DBI_username	'webuser'
 PerlSetVar Auth_DBI_password	'webpass'
 PerlSetVar Auth_DBI_pwd_table	'SysUser'
 PerlSetVar Auth_DBI_uid_field	'email'
 PerlSetVar Auth_DBI_pwd_field	'password'
 PerlSetVar Auth_DBI_encrypted	'off'
 require valid-user
 AllowOverride None
