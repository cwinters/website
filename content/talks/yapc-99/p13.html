Simple SELECT example

Simple example of a SELECT and fetch:

 my $username = $cgi->remote_user();
 my $sql = qq/
   SELECT f_name, l_name, last_login
     FROM userlist
    WHERE username = '$username'
 /;
 my ( $sth );
 eval {
  $sth = $dbh->prepare( $sql );
  $sth->execute;
 };
 die "Cannot fetch user data for $username: $@" if ( $@ );

 my $row = $sth->fetchrow_arrayref;
 print $cgi->p( 'Hello ', $row->[0], $row->[1], '. It appears ',
                'the last time I saw you was ', $row->[2] );
