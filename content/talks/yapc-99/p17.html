Binding parameters: UPDATE example

 my $inv = $cgi->param( 'widget_inventor' );
 my $ref = $cgi->param( 'widget_ref' );
 my $sql = qq/
   UPDATE widget_info
      SET widget_inventor = ?,
          widget_ref = ?
    WHERE widget_id = $widget_id
 /;
 my ( $rv );
 eval {
  $dbh->do( $sql, undef, $inv, $ref );
 };
 if ( $@ ) { 
   my $err = $@ || $DBI::errstr;
   die "Cannot update with values $inv, $ref. Error: $err";
 }
