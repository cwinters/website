Common package actions

Package usage examples:

# Shortcuts

 $ export OPENINTERACT=/opt/OpenInteract
 $ export OIWEBSITE=/home/httpd/mysite.com

# Create a new package

 $ oi_manage create_skeleton --package=mypkg

# Check and export it

 $ oi_manage check_package
 $ oi_manage export_package

# Install it to the base

 $ oi_manage install_package --package_file=mypkg-0.01.tar.gz

# Create a new website

 $ oi_manage --website_name=MySite

# Install my package to the website

 $ oi_manage apply_package --package=mypkg

# Install SQL for all initial packages and mine

 $ oi_manage install_sql --package=INITIAL,mypkg

# Install templates for initial packages and mine

 $ oi_manage install_template --package=INITIAL,mypkg

# Change the SPOPS driver

 $ oi_manage change_spops_driver --driver=SPOPS::DBI::Pg
