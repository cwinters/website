Object configuration

Sample object configuration

$spops = {
 'user' => {
    class        => 'My::User',
    code_class   => 'My::UserCode',
    isa          => [ qw/ SPOPS::Secure SPOPS::DBI::Pg  SPOPS::DBI / ],
    field        => [ qw/ user_id first_name last_name email language
                          title last_login num_logins login_name password
                          removal_date notes theme_id / ],
    id_field     => 'user_id',
    increment_field => 1,
    no_insert    => [ qw/ user_id / ],
    skip_undef   => [ qw/ last_login num_logins password language 
                          theme_id / ],
    no_update    => [ qw/ user_id / ],
    key_table    => 'sys_user',
    base_table   => 'sys_user',
    sql_defaults => [ qw/ language theme_id / ],
    alias        => [],
    has_a        => { 'My::Theme' => [ 'theme_id' ], },
    links_to     => { 'My::Group' => 'sys_group_user' },
    fetch_by     => [ 'login_name' ],
    creation_security => { 
       u => undef,
       g   => { 3 => 'WRITE' },
       w   => 'READ',
    },
    track        => { create => 0, update => 1, remove => 1 },
    display      => { url => '/User/show/' },
    name         => sub { return $_[0]->full_name },
    object_name  => 'User',
  },
};
