Simple SPOPS example

# Read in your configuration file and create the classes

SPOPS::Initialize->process({ filename => 'my_config.perl' });

# Create a new user

my $user = My::User->new();
$user->{login_name} = 'yapc';
$user->{password}   = SPOPS::Utility->crypt_it( 'montreal' );
my $user_id = eval { $user->save };
die "Caught an error saving user: $@"  if ( $@ );
print "New user ID: $user_id\n";

# Fetch lots of users and display

my $user_list = eval { My::User->fetch_group({ 
                                 where => "last_name like ?",
                                 value => [ 'wa%' ] }) };
foreach my $user ( @{ $user_list } ) {
    printf( "User %s is %s\n", $user->{login_name}, $user->full_name );
}

# Fetch groups this user belongs to
my $group_list = eval { $user->group };
foreach my $group ( @{ $group_list } ) {
    print "Group: $group->{name}\n";
}
